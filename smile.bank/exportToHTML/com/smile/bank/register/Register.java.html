67   					if (success == 2) {
68   						break;
69   					}

71   				}

73   				smile.message("Credential Creation Failed!");
74   				smile.eventFail(e);
75   				fail = 1;
76   				// problem here is that customer info is saved but credentials are not
77   				// this solves that by deleting customer by id from table
78   
79   				if (fail == 1) {
80   					try {
81   						regcustServ.purge(1, ID);
82   
83   					} catch (SmileException e1) {
84   						e1.printStackTrace();
85   					}

87   				}
88   			}
89   		}

91   		if (success == 2) {
